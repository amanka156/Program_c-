Потоки введення/виведення

C++ використовує потокову модель для роботи з файлами. Основні класи:

КласОписЗаголовок
ofstreamВихідний файловий потік (запис)<fstream>
ifstreamВхідний файловий потік (читання)<fstream>
fstreamДвонаправлений потік<fstream>

fstream — робота з файлами

Підключення та оголошення
#include <fstream>
using namespace std;

// Запис
ofstream outFile("data.txt");

// Читання
ifstream inFile("data.txt");

// Двонаправлений
fstream ioFile("data.txt");

Запис у файл

Запис
#include <fstream>
#include <iostream>
using namespace std;

int main() {
    ofstream fout("output.txt");
    
    if (!fout) {
        cout << "Помилка відкриття файлу!" << endl;
        return 1;
    }
    
    fout << "Привіт, світе!" << endl;
    fout << "Це перший рядок." << endl;
    fout << "Число: " << 42 << endl;
    
    fout.close();
    cout << "Дані записано у output.txt" << endl;
    
    return 0;
}

Читання з файлу

Читання
#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int main() {
    ifstream fin("output.txt");
    
    if (!fin) {
        cout << "Помилка відкриття файлу!" << endl;
        return 1;
    }
    
    // Спосіб 1: читання рядка за рядком
    string line;
    while (getline(fin, line)) {
        cout << line << endl;
    }
    
    // Спосіб 2: читання слова за словом
    string word;
    while (fin >> word) {
        cout << word << endl;
    }
    
    // Спосіб 3: читання символ за символом
    char ch;
    while (fin.get(ch)) {
        cout << ch;
    }
    
    fin.close();
    return 0;
}

Режими відкриття файлу

РежимОпис
ios::inВідкриття для читання
ios::outВідкриття для запису
ios::appДодавання в кінець файлу
ios::truncВидалення вмісту при відкритті
ios::binaryДвійковий режим
ios::ateПерехід в кінець після відкриття
Режими
ofstream fout("data.txt", ios::out | ios::app);   // додавання
ifstream fin("data.txt", ios::in);                 // читання
fstream io("data.txt", ios::in | ios::out);        // читання + запис

Перевірка стану файлу

Перевірка
ifstream fin("data.txt");

if (!fin) {                    // або if (!fin.is_open())
    cout << "Файл не відкрито!" << endl;
}

if (fin.eof()) {               // кінець файлу
    cout << "Кінець файлу" << endl;
}

if (fin.fail()) {              // помилка операції
    cout << "Помилка!" << endl;
}

if (fin.good()) {              // все добре
    cout << "OK" << endl;
}

Двійкові файли

Двійковий запис/читання
// Запис структури у двійковий файл
struct Student {
    char name[50];
    int age;
    double grade;
};

// Запис
ofstream fout("students.bin", ios::binary);
Student s1 = {"Ivan", 20, 4.5};
fout.write(reinterpret_cast<char*>(&s1), sizeof(s1));
fout.close();

// Читання
ifstream fin("students.bin", ios::binary);
Student s2;
fin.read(reinterpret_cast<char*>(&s2), sizeof(s2));
cout << s2.name << " " << s2.age << " " << s2.grade << endl;
fin.close();

💻 Практичні завдання

1Копіювання файлу

Напишіть програму, яка копіює вміст одного текстового файлу в інший.

2Підрахунок рядків

Напишіть програму, яка підраховує кількість рядків, слів та символів у текстовому файлі.

3Пошук у файлі

Напишіть програму, яка шукає задане слово у файлі та виводить номери рядків, де воно зустрічається.

4Заміна у файлі

Створіть програму, яка замінює всі входження слова A на слово B у файлі.

5База даних студентів у файлі

Зберігайте масив структур Student у двійковому файлі. Реалізуйте додавання, читання, пошук.

6Логування

Створіть функцію log(message), яка додає повідомлення з часом у файл log.txt.

7Злиття файлів

Напишіть програму, яка зливає два відсортованих текстових файли в один відсортований.

8CSV парсер

Напишіть програму, яка зчитує CSV файл та виводить дані у вигляді відформатованої таблиці.

9Шифрування файлу

Реалізуйте шифр Цезаря для всього файлу: зчитати файл, зашифрувати, записати результат.

10Розбиття файлу

Напишіть програму, яка розбиває великий файл на частини заданого розміру (наприклад, для передачі).
