Масив вказівників

Масив вказівників — це масив, елементами якого є вказівники. Кожен елемент може вказувати на окремий об'єкт у пам'яті.

Масив вказівників
int a = 10, b = 20, c = 30;
int* arr[3] = {&a, &b, &c};

// Доступ до значень
cout << *arr[0] << endl;  // 10
cout << *arr[1] << endl;  // 20
cout << *arr[2] << endl;  // 30

// Зміна значення
*arr[1] = 100;            // b = 100

Вказівник на вказівник

Вказівник на вказівник
int x = 42;
int* p = &x;       // вказівник на x
int** pp = &p;     // вказівник на вказівник

cout << x << endl;    // 42
cout << *p << endl;   // 42
cout << **pp << endl; // 42

**pp = 100;           // x = 100

Вказівник на функцію

Вказівник на функцію зберігає адресу функції і дозволяє викликати її через цей вказівник.

Вказівник на функцію
// Оголошення: тип_повернення (*ім'я)(параметри);
int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }

int main() {
    // Вказівник на функцію
    int (*operation)(int, int);
    
    operation = add;
    cout << operation(3, 4) << endl;  // 7
    
    operation = sub;
    cout << operation(3, 4) << endl;  // -1
    
    operation = mul;
    cout << operation(3, 4) << endl;  // 12
    
    return 0;
}

Функції зворотного виклику (callbacks)

Callback
// Функція приймає вказівник на функцію
void processArray(int arr[], int n, int (*func)(int)) {
    for (int i = 0; i < n; i++)
        arr[i] = func(arr[i]);
}

int square(int x) { return x * x; }
int cube(int x) { return x * x * x; }

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    
    processArray(arr, 5, square);  // підносить до квадрата
    processArray(arr, 5, cube);    // підносить до куба
    
    return 0;
}
