Масив структур

Масив структур — це масив, елементами якого є структури. Це дозволяє зберігати та обробляти набір об'єктів одного типу (наприклад, список студентів, каталог товарів).

Масив структур
struct Student {
    string name;
    int age;
    double avgGrade;
};

// Масив з 100 студентів
Student group[100];

// Доступ до членів
group[0].name = "Іван";
group[0].age = 20;
group[0].avgGrade = 4.5;

// Ініціалізація
Student smallGroup[3] = {
    {"Anna", 19, 4.8},
    {"Boris", 20, 3.9},
    {"Clara", 19, 4.2}
};

Введення/виведення
void inputStudents(Student arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << "Студент #" << i + 1 << endl;
        cout << "Ім'я: ";
        cin.ignore();
        getline(cin, arr[i].name);
        cout << "Вік: ";
        cin >> arr[i].age;
        cout << "Середній бал: ";
        cin >> arr[i].avgGrade;
    }
}

void printStudents(const Student arr[], int n) {
    cout << "\n--- Список студентів ---" << endl;
    for (int i = 0; i < n; i++) {
        cout << i + 1 << ". " << arr[i].name
             << ", " << arr[i].age << " років"
             << ", бал: " << arr[i].avgGrade << endl;
    }
}

Пошук та фільтрація

Пошук
// Пошук за іменем
int findByName(const Student arr[], int n, const string& name) {
    for (int i = 0; i < n; i++)
        if (arr[i].name == name)
            return i;
    return -1;
}

// Фільтрація за критерієм
void printExcellent(const Student arr[], int n) {
    cout << "Відмінники (бал >= 4.5):" << endl;
    for (int i = 0; i < n; i++)
        if (arr[i].avgGrade >= 4.5)
            cout << arr[i].name << endl;
}

// Пошук максимуму
int findBestStudent(const Student arr[], int n) {
    int bestIdx = 0;
    for (int i = 1; i < n; i++)
        if (arr[i].avgGrade > arr[bestIdx].avgGrade)
            bestIdx = i;
    return bestIdx;
}

Сортування масиву структур

Сортування
// Сортування за балом (спадання) — Bubble Sort
void sortByGrade(Student arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].avgGrade < arr[j+1].avgGrade) {
                Student tmp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = tmp;
            }
}

// Сортування за ім'ям
void sortByName(Student arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].name > arr[j+1].name) {
                Student tmp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = tmp;
            }
}

Збереження масиву структур у файл

Робота з файлом
#include <fstream>

void saveToFile(const Student arr[], int n, const string& filename) {
    ofstream fout(filename);
    fout << n << endl;
    for (int i = 0; i < n; i++) {
        fout << arr[i].name << endl;
        fout << arr[i].age << endl;
        fout << arr[i].avgGrade << endl;
    }
    fout.close();
}

void loadFromFile(Student arr[], int& n, const string& filename) {
    ifstream fin(filename);
    fin >> n;
    fin.ignore();
    for (int i = 0; i < n; i++) {
        getline(fin, arr[i].name);
        fin >> arr[i].age;
        fin >> arr[i].avgGrade;
        fin.ignore();
    }
    fin.close();
}

💻 Практичні завдання

1Бібліотека книг

Створіть масив з 50 книг (Book: назва, автор, рік, жанр). Реалізуйте пошук за автором, фільтрацію за роком, сортування за назвою.

2Залізничні квитки

Створіть масив квитків (номер, пункт відправлення, пункт прибуття, дата, ціна). Пошук за напрямком, сортування за ціною.

3Електронний журнал

Створіть журнал оцінок для 30 студентів з 5 предметів. Обчисліть середній бал кожного студента та кожного предмета.

4Інвентаризація складу

Створіть масив товарів (назва, артикул, кількість, ціна). Функції: додавання, видалення, пошук за артикулом, загальна вартість.

5Спортивні змагання

Створіть масив команд (назва, кількість ігор, перемог, нічиїх, поразок, очки). Сортування за очками, визначення чемпіону.

6Записна книжка

Створіть масив контактів (ім'я, телефон, email, адреса). Пошук за іменем, сортування, збереження/завантаження з файлу.

7Автосалон

Створіть масив автомобілів (марка, модель, рік, ціна, пробіг). Фільтрація за ціною, сортування за роком випуску.

8Аптека

Створіть масив ліків (назва, виробник, ціна, термін придатності, кількість). Пошук за назвою, видалення прострочених, сортування за ціною.

9Кінотеатр

Створіть масив фільмів (назва, режисер, рік, жанр, рейтинг). Топ-10 за рейтингом, пошук за жанром, сортування за роком.

10Університет

Створіть систему обліку студентів: додавання, видалення, редагування, пошук, сортування за різними критеріями, збереження у файл.
